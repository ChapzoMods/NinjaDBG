    79→addr_t HeadlessCLI::parseAddr(const std::string& s, bool* ok) {
    80→    errno = 0;
    81→    char* end = nullptr;
    82→    unsigned long long v = strtoull(s.c_str(), &end, 0);
    83→    if (ok) *ok = (errno == 0 && end != s.c_str() && *end == '\0');
    84→    return (addr_t)v;
    85→}

    87→std::vector<u8> HeadlessCLI::parseBytes(const std::vector<std::string>& tokens) {
    88→    std::vector<u8> bytes;
    89→    for (auto& t : tokens) {
    90→        u8 b = (u8)strtoul(t.c_str(), nullptr, 0);
    91→        bytes.push_back(b);
    92→    }
    93→    return bytes;
    94→}

    96→void HeadlessCLI::execute(const std::string& line) {
    97→    std::istringstream ss(line);
    98→    std::string cmd;
    99→    ss >> cmd;
   100→    if (cmd.empty()) return;
   101→
   102→    std::vector<std::string> args;
   103→    std::string a;
   104→    while (ss >> a) args.push_back(a);
   105→
   106→    if (cmd == "help" || cmd == "h" || cmd == "?")      cmdHelp(args);
   107→    else if (cmd == "quit" || cmd == "q" || cmd == "exit") cmdQuit(args);
   108→    else if (cmd == "attach")    cmdAttach(args);
   109→    else if (cmd == "launch")    cmdLaunch(args);
   110→    else if (cmd == "detach")    cmdDetach(args);
   111→    else if (cmd == "kill")      cmdKill(args);
   112→    else if (cmd == "continue" || cmd == "cont" || cmd == "c") cmdContinue(args);
   113→    else if (cmd == "step" || cmd == "s" || cmd == "stepi" || cmd == "si") cmdStep(args);
   114→    else if (cmd == "next" || cmd == "n") cmdNext(args);
   115→    else if (cmd == "finish" || cmd == "fo") {
   116→        if (dbg_.pid() == 0) { err("Not attached"); return; }
   117→        if (dbg_.stepOut()) out("Stepped out of current function");
   118→        else err("step-out failed: " + dbg_.lastError());
   119→    }
   120→    else if (cmd == "break" || cmd == "b")  cmdBreak(args, false);
   121→    else if (cmd == "tbreak" || cmd == "tb") cmdBreak(args, true);
   122→    else if (cmd == "delete" || cmd == "d")  cmdDelete(args);
   123→    else if (cmd == "watch") {
   124→        // watch <addr> [len] [w|rw|x]
   125→        if (args.size() < 1) { err("usage: watch <addr> [len] [w|rw|x]"); return; }
   126→        bool ok; addr_t a = parseAddr(args[0], &ok);
   127→        if (!ok) { err("bad address"); return; }
   128→        size_t len = args.size() > 1 ? strtoul(args[1].c_str(), nullptr, 0) : 8;
   129→        DebuggerCore::WatchType wt = DebuggerCore::WatchType::Write;
   130→        if (args.size() > 2) {
   131→            if (args[2] == "rw") wt = DebuggerCore::WatchType::ReadWrite;
   132→            else if (args[2] == "x") wt = DebuggerCore::WatchType::Execute;
   133→        }
   134→        int id = dbg_.addWatchpoint(a, len, wt);
   135→        out("Watchpoint " + std::to_string(id) + " set at " + hex(a));
   136→    }
   137→    else if (cmd == "info" || cmd == "i") cmdInfo(args);
   138→    else if (cmd == "x") cmdExamine(args);
   139→    else if (cmd == "set") cmdSet(args);
   140→    else if (cmd == "disas" || cmd == "dis") cmdDisas(args);
   141→    else if (cmd == "edit")   cmdEdit(args);
   142→    else if (cmd == "decomp" || cmd == "dec") cmdDecomp(args);
   143→    else if (cmd == "pretty" || cmd == "pp") cmdPretty(args);
   144→    else if (cmd == "script") cmdScript(args);
   145→    else if (cmd == "backtrace" || cmd == "bt") cmdBacktrace(args);
   146→    else if (cmd == "patch") cmdPatch(args);
   147→    else if (cmd == "stealth") cmdStealth(args);
   148→    else if (cmd == "kernel") cmdKernel(args);
   149→    else if (cmd == "target") cmdTarget(args);
   150→    else if (cmd == "syscall-step") {
   151→        int nr; bool entry;
   152→        if (dbg_.stepToSyscall(nr, entry)) {
   153→            out(std::string("Syscall ") + (entry ? "entry" : "exit") + " nr=" + std::to_string(nr));
   154→        } else err("syscall-step failed");
   155→    }
   156→    else {
   157→        err("Unknown command: " + cmd + " (type 'help')");
   158→    }
   159→}

   163→void HeadlessCLI::cmdAttach(const std::vector<std::string>& args) {
   164→    if (args.empty()) { err("usage: attach <pid>"); return; }
   165→    pid_t p = (pid_t)atoi(args[0].c_str());
   166→    if (dbg_.attach(p)) {
   167→        out("Attached to pid " + std::to_string(p));
   168→        dbg_.readMaps();  // refresh cache
   169→    } else {
   170→        err("Attach failed: " + dbg_.lastError());
   171→    }
   172→}

   214→void HeadlessCLI::cmdBreak(const std::vector<std::string>& args, bool temporary) {
   215→    if (args.empty()) { err("usage: break <addr> [condition]"); return; }
   216→    bool ok; addr_t a = parseAddr(args[0], &ok);
   217→    if (!ok) { err("bad address"); return; }
   218→    int id;
   219→    std::string cond;
   220→    if (args.size() > 1) {
   221→        for (size_t i = 1; i < args.size(); i++) {
   222→            if (i > 1) cond += " ";
   223→            cond += args[i];
   224→        }
   225→    }
   226→    if (cond.empty()) {
   227→        id = temporary ? dbg_.addTempBreakpoint(a) : dbg_.addBreakpoint(a);
   228→    } else {
   229→        id = dbg_.addConditionalBreakpoint(a, cond);
   230→    }
   231→    if (id < 0) err("set breakpoint failed");
   232→    else out("Breakpoint " + std::to_string(id) + " set at " + hex(a) +
   233→             (temporary ? " (temporary)" : "") +
   234→             (cond.empty() ? "" : " when " + cond));
   235→}

   257→void HeadlessCLI::cmdExamine(const std::vector<std::string>& args) {
   258→    // x /Nxb <addr>
   259→    if (args.size() < 2) { err("usage: x /Nxb <addr>  (or  x /Nxw <addr>)"); return; }
   260→    std::string fmt = args[0];  // /Nxb or /Nxw
   261→    bool ok; addr_t a = parseAddr(args[1], &ok);
   262→    if (!ok) { err("bad address"); return; }
   263→    // Parse /Nxb or /Nxw
   264→    int count = 16; int size = 1; bool word = false;
   265→    if (fmt.size() > 1 && fmt[0] == '/') {
   266→        // Find x
   267→        size_t xpos = fmt.find('x');
   268→        if (xpos != std::string::npos) {
   269→            count = atoi(fmt.substr(1, xpos - 1).c_str());
   270→            if (count <= 0) count = 16;
   271→            if (xpos + 1 < fmt.size()) {
   272→                char c = fmt[xpos + 1];
   273→                if (c == 'w') { size = 4; word = true; }
   274→                else if (c == 'g') { size = 8; word = true; }
   275→                else if (c == 'h') { size = 2; word = true; }
   276→            }
   277→        }
   278→    }
   279→    auto bytes = dbg_.readMemoryVec(a, count * size);
   280→    if (bytes.empty()) { err("read failed"); return; }
   281→    for (int i = 0; i < count; i++) {
   282→        if (i % 16 == 0) {
   283→            if (i > 0) std::cout << "\n";
   284→            std::cout << hex(a + i * size) << ": ";
   285→        }
   286→        if (word) {
   287→            u64 v = 0;
   288→            std::memcpy(&v, bytes.data() + i * size, size);
   289→            std::cout << hex(v, size * 2) << " ";
   290→        } else {
   291→            std::cout << hex2(bytes[i], 2) << " ";
   292→        }
   293→    }
   294→    std::cout << std::endl;
   295→}

   297→void HeadlessCLI::cmdSet(const std::vector<std::string>& args) {
   298→    // set <addr> = <bytes>
   299→    if (args.size() < 3 || args[1] != "=") {
   300→        err("usage: set <addr> = <byte1> <byte2> ...");
   301→        return;
   302→    }
   303→    bool ok; addr_t a = parseAddr(args[0], &ok);
   304→    if (!ok) { err("bad address"); return; }
   305→    std::vector<std::string> btokens(args.begin() + 2, args.end());
   306→    auto bytes = parseBytes(btokens);
   307→    if (dbg_.writeMemory(a, bytes.data(), bytes.size())) {
   308→        out("Wrote " + std::to_string(bytes.size()) + " bytes at " + hex(a));
   309→    } else err("write failed");
   310→}

   312→void HeadlessCLI::cmdDisas(const std::vector<std::string>& args) {
   313→    if (dbg_.pid() == 0) { err("Not attached. Use 'attach <pid>' first."); return; }
   314→    addr_t addr = 0;
   315→    size_t count = 20;
   316→    // If no args, default to current RIP
   317→    if (args.empty()) {
   318→        RegisterSet r;
   319→        if (!dbg_.readRegisters(r)) { err("readRegisters failed"); return; }
   320→        addr = r.rip;
   321→    } else {
   322→        bool ok; addr = parseAddr(args[0], &ok);
   323→        if (!ok) { err("bad address"); return; }
   324→        if (args.size() > 1) count = (size_t)strtoul(args[1].c_str(), nullptr, 0);
   325→    }
   326→    // Read enough bytes
   327→    auto bytes = dbg_.readMemoryVec(addr, count * 15 + 32);
   328→    if (bytes.empty()) { err("readMemory failed at " + hex(addr)); return; }
   329→    auto instrs = disas_.disassemble(addr, bytes.data(), bytes.size(), count);
   330→    if (instrs.empty()) { err("no instructions decoded"); return; }
   331→    // Header
   332→    std::cout << "ADDRESS             BYTES               MNEMONIC OPERANDS\n";
   333→    std::cout << "------------------- ------------------- ------------------------------\n";
   334→    RegisterSet regs;
   335→    bool have_regs = dbg_.readRegisters(regs);
   336→    for (auto& ins : instrs) {
   337→        bool at_rip = have_regs && (ins.address == regs.rip);
   338→        std::string line = Disassembler::format(ins, true);
   339→        // Annotate with symbol if it's a call/jmp
   340→        if (ins.has_branch_target && ins.branch_target != 0) {
   341→            std::string sym = resolveSymbol(ins.branch_target);
   342→            if (!sym.empty()) {
   343→                line += "  ; " + sym;
   344→            }
   345→        }
   346→        if (at_rip) {
   347→            std::cout << "\x1b[33m>> " << line << "\x1b[0m\n";  // yellow highlight
   348→        } else {
   349→            std::cout << "   " << line << "\n";
   350→        }
   351→    }
   352→}

   354→void HeadlessCLI::cmdEdit(const std::vector<std::string>& args) {
   355→    if (dbg_.pid() == 0) { err("Not attached. Use 'attach <pid>' first."); return; }
   356→    addr_t addr = 0;

   358→        RegisterSet r;
   359→        if (!dbg_.readRegisters(r)) { err("readRegisters failed"); return; }
   360→        addr = r.rip;

   368→    editor_->run(addr, InteractiveMemoryEditor::Mode::Hex);
   369→    editor_.reset();
   370→    out("Returned from editor.");
   371→}

   373→void HeadlessCLI::cmdDecomp(const std::vector<std::string>& args) {
   374→    if (args.empty()) {
   375→        // No args: decompile function at current RIP (live process)
   376→        if (dbg_.pid() == 0) { printDecompStatus(); return; }
   377→        RegisterSet r;
   378→        if (!dbg_.readRegisters(r)) { err("readRegisters failed"); return; }
   379→        out("Decompiling function at current RIP = " + hex(r.rip) + " ...");
   380→        auto res = decompiler_.decompileFunction(dbg_, r.rip, 0x1000);
   381→        if (!res.ok) {
   382→            err("Decompile failed: " + res.error);
   383→            if (res.error.find("not available") != std::string::npos ||
   384→                res.error.find("No decompiler") != std::string::npos) {
   385→                err("Run 'decomp list' to see backend status, or 'decomp api' for install hints.");
   386→            }
   387→            return;
   388→        }
   389→        out("Backend: " + res.backend_used + "  Elapsed: " + std::to_string((int)res.elapsed_ms) + " ms");
   390→        out("Function: " + res.function_name);
   391→        std::cout << "\n" << res.c_code << std::endl;
   392→        return;
   393→    }
   394→    std::string sub = args[0];
   395→
   396→    // decomp list
   397→    if (sub == "list") {
   398→        printDecompStatus();
   399→        return;
   400→    }
   401→    // decomp api
   402→    if (sub == "api") {
   403→        std::cout << Decompiler::apiDocs() << std::endl;
   404→        return;
   405→    }
   406→    // decomp set <backend>
   407→    if (sub == "set") {
   408→        if (args.size() < 2) { err("usage: decomp set <retdec-native|retdec-subprocess|angr|auto>"); return; }
   409→        std::string b = args[1];
   410→        if (b == "auto")              decompiler_.setBackend(Decompiler::Backend::Auto);
   411→        else if (b == "retdec-native")decompiler_.setBackend(Decompiler::Backend::RetDecNative);
   412→        else if (b == "retdec-subprocess") decompiler_.setBackend(Decompiler::Backend::RetDecSubprocess);
   413→        else if (b == "retdec")       decompiler_.setBackend(Decompiler::Backend::RetDecNative);
   414→        else if (b == "angr")         decompiler_.setBackend(Decompiler::Backend::Angr);
   415→        else { err("unknown backend: " + b); return; }
   416→        out("Decompiler backend set to: " + b);
   417→        return;
   418→    }
   419→    // decomp file <binary> [addr]
   420→    if (sub == "file") {
   421→        if (args.size() < 2) {
   422→            err("usage: decomp file <binary> [addr]");
   423→            return;
   424→        }
   425→        std::string bin = args[1];
   426→        if (args.size() >= 3) {
   427→            // Decompile one function in a file
   428→            bool ok; addr_t a = parseAddr(args[2], &ok);
   429→            if (!ok) { err("bad address: " + args[2]); return; }
   430→            out("Decompiling function at " + hex(a) + " in " + bin + " ...");
   431→            auto r = decompiler_.decompileFunctionInFile(bin, a);
   432→            if (!r.ok) { err("Decompile failed: " + r.error); return; }
   433→            out("Backend: " + r.backend_used + "  Elapsed: " + std::to_string((int)r.elapsed_ms) + " ms");
   434→            out("Function: " + r.function_name);
   435→            std::cout << "\n" << r.c_code << std::endl;
   436→        } else {
   437→            // Decompile whole file
   438→            out("Decompiling entire file: " + bin + " ...");
   439→            auto r = decompiler_.decompileFile(bin);
   440→            if (!r.ok) { err("Decompile failed: " + r.error); return; }
   441→            out("Backend: " + r.backend_used + "  Elapsed: " + std::to_string((int)r.elapsed_ms) + " ms");
   442→            std::cout << "\n" << r.c_code << std::endl;
   443→        }
   444→        return;
   445→    }
   446→
   447→    // decomp <addr> [max_bytes]  - decompile function in live process
   448→    bool ok; addr_t addr = parseAddr(sub, &ok);
   449→    if (!ok) { err("unknown decomp subcommand: " + sub + "\n  usage: decomp [addr|file|list|api|set]"); return; }
   450→    if (dbg_.pid() == 0) { err("Not attached. Use 'attach <pid>' first."); return; }
   451→    size_t max_bytes = 0x1000;
   452→    if (args.size() > 1) {
   453→        max_bytes = (size_t)strtoull(args[1].c_str(), nullptr, 0);
   454→        if (max_bytes == 0) max_bytes = 0x1000;
   455→    }
   456→    out("Decompiling function at " + hex(addr) + " (max " + std::to_string(max_bytes) + " bytes) ...");
   457→    auto r = decompiler_.decompileFunction(dbg_, addr, max_bytes);
   458→    if (!r.ok) {
   459→        err("Decompile failed: " + r.error);
   460→        if (r.error.find("not available") != std::string::npos ||
   461→            r.error.find("No decompiler") != std::string::npos) {
   462→            err("Run 'decomp list' to see backend status, or 'decomp api' for install hints.");

   493→void HeadlessCLI::cmdPretty(const std::vector<std::string>& args) {
   494→    if (args.empty()) {
   495→        out("Active language: " + PrettyPrinter::languageName(pretty_.currentLanguage()));
   496→        out("Available: c, cpp, rust, go, python");
   497→        out("Use 'pretty api' for full documentation.");

   499→    }
   500→    std::string sub = args[0];
   501→
   502→    if (sub == "list") {
   503→        out("Pretty printers available:");
   504→        for (auto l : PrettyPrinter::allLanguages()) {
   505→            out("  " + PrettyPrinter::languageName(l));
   506→        }
   507→        out("Current: " + PrettyPrinter::languageName(pretty_.currentLanguage()));
   508→        return;
   509→    }
   510→    if (sub == "api") {
   511→        std::cout << PrettyPrinter::apiDocs() << std::endl;
   512→        return;
   513→    }
   514→    if (sub == "set") {
   515→        if (args.size() < 2) { err("usage: pretty set <c|cpp|rust|go|python|none>"); return; }
   516→        std::string lang = args[1];
   517→        if (lang == "c" || lang == "C")             pretty_.setLanguage(PrettyPrinter::Language::C);
   518→        else if (lang == "cpp" || lang == "c++" || lang == "C++") pretty_.setLanguage(PrettyPrinter::Language::Cpp);
   519→        else if (lang == "rust" || lang == "rs")    pretty_.setLanguage(PrettyPrinter::Language::Rust);
   520→        else if (lang == "go" || lang == "golang")  pretty_.setLanguage(PrettyPrinter::Language::Go);
   521→        else if (lang == "python" || lang == "py")  pretty_.setLanguage(PrettyPrinter::Language::Python);
   522→        else if (lang == "none" || lang == "off")   pretty_.setLanguage(PrettyPrinter::Language::NoLanguage);
   523→        else { err("unknown language: " + lang); return; }
   524→        out("Pretty printer language set to: " + lang);
   525→        return;
   526→    }
   527→
   528→    // All remaining subcommands need an attached process
   529→    if (dbg_.pid() == 0) { err("Not attached. Use 'attach <pid>' first."); return; }
   530→
   531→    if (sub == "auto") {
   532→        if (args.size() < 2) { err("usage: pretty auto <addr>"); return; }
   533→        bool ok; addr_t a = parseAddr(args[1], &ok);
   534→        if (!ok) { err("bad address"); return; }
   535→        out(pretty_.autoPrint(dbg_, a));
   536→        return;
   537→    }
   538→    if (sub == "cstring") {
   539→        if (args.size() < 2) { err("usage: pretty cstring <addr>"); return; }
   540→        bool ok; addr_t a = parseAddr(args[1], &ok);
   541→        if (!ok) { err("bad address"); return; }
   542→        out(pretty_.printCString(dbg_, a));
   543→        return;
   544→    }
   545→    if (sub == "cpp_string" || sub == "cppstring" || sub == "stdstring") {
   546→        if (args.size() < 2) { err("usage: pretty cpp_string <addr>"); return; }
   547→        bool ok; addr_t a = parseAddr(args[1], &ok);
   548→        if (!ok) { err("bad address"); return; }
   549→        out(pretty_.printCppString(dbg_, a));
   550→        return;
   551→    }
   552→    if (sub == "rust_string" || sub == "ruststring") {
   553→        if (args.size() < 2) { err("usage: pretty rust_string <addr>"); return; }
   554→        bool ok; addr_t a = parseAddr(args[1], &ok);
   555→        if (!ok) { err("bad address"); return; }
   556→        out(pretty_.printRustString(dbg_, a));
   557→        return;
   558→    }
   559→    if (sub == "go_string" || sub == "gostring") {
   560→        if (args.size() < 2) { err("usage: pretty go_string <addr>"); return; }
   561→        bool ok; addr_t a = parseAddr(args[1], &ok);
   562→        if (!ok) { err("bad address"); return; }
   563→        out(pretty_.printGoString(dbg_, a));
   564→        return;
   565→    }
   566→    if (sub == "py_string" || sub == "pystring" || sub == "python_string") {
   567→        if (args.size() < 2) { err("usage: pretty py_string <addr>"); return; }
   568→        bool ok; addr_t a = parseAddr(args[1], &ok);
   569→        if (!ok) { err("bad address"); return; }
   570→        out(pretty_.printPyString(dbg_, a));
   571→        return;

   573→    if (sub == "struct") {
   574→        if (args.size() < 3) {
   575→            err("usage: pretty struct <addr> <descriptor>");
   576→            err("  descriptor: comma-separated types, e.g. i32,str,ptr,u64");
   577→            err("  valid types: i8 i16 i32 i64 u8 u16 u32 u64 f32 f64 ptr str hex<N>");

   581→        if (!ok) { err("bad address"); return; }
   582→        // Reconstruct descriptor (may contain commas that were split)
   583→        std::string desc;
   584→        for (size_t i = 2; i < args.size(); i++) {
   585→            if (i > 2) desc += ",";
   586→            desc += args[i];
   587→        }
   588→        std::cout << pretty_.printStruct(dbg_, a, desc) << std::endl;
   589→        return;
   590→    }
   591→    err("unknown pretty subcommand: " + sub + " (valid: set|list|api|auto|cstring|cpp_string|rust_string|go_string|py_string|struct)");
   592→}

   594→void HeadlessCLI::cmdScript(const std::vector<std::string>& args) {
   595→    if (args.empty()) { printScriptStatus(); return; }
   596→    std::string sub = args[0];
   597→    if (sub == "list") {
   598→        printScriptStatus();
   599→    } else if (sub == "api") {
   600→        std::cout << ScriptEngine::apiDocs() << std::endl;
   601→    } else if (sub == "run") {
   602→        if (args.size() < 3) {
   603→            err("usage: script run <lua|python> <file-or-code>");
   604→            err("  If the third arg starts with a path separator or ./, treat");
   605→            err("  as a file. Otherwise treat as inline code.");
   606→            return;
   607→        }
   608→        std::string lang_str = args[1];
   609→        std::string payload = args[2];
   610→        ScriptEngine::Lang lang;
   611→        if (lang_str == "lua" || lang_str == "Lua") lang = ScriptEngine::Lang::Lua;
   612→        else if (lang_str == "python" || lang_str == "py" || lang_str == "Python") lang = ScriptEngine::Lang::Python;
   613→        else { err("Unknown language: " + lang_str + " (use 'lua' or 'python')"); return; }
   614→
   615→        if (!script_->isAvailable(lang)) {
   616→            err(script_->installHint(lang));
   617→            return;
   618→        }
   619→
   620→        // If payload looks like a file path that exists, read it; else treat as code
   621→        std::string code;
   622→        std::ifstream f(payload);
   623→        if (f) {
   624→            std::stringstream ss; ss << f.rdbuf();
   625→            code = ss.str();
   626→            out("Running " + lang_str + " script: " + payload);
   627→        } else {
   628→            code = payload;
   629→            out("Running inline " + lang_str + " code");
   630→        }
   631→
   632→        std::string err_str;
   633→        if (script_->runString(lang, code, err_str)) {
   634→            out("Script completed.");
   635→        } else {
   636→            err("Script error: " + err_str);
   637→        }
   638→    } else {
   639→        err("usage: script <list|run|api>");
   640→    }
   641→}

   643→std::string HeadlessCLI::resolveSymbol(addr_t a) {
   644→    auto maps = dbg_.readMaps();
   645→    for (auto& m : maps) {
   646→        if (a >= m.start && a < m.end) {
   647→            size_t slash = m.path.rfind('/');
   648→            std::string bn = (slash != std::string::npos) ? m.path.substr(slash + 1) : m.path;
   649→            return "<" + bn + "+0x" + hex2(a - m.start, 0) + ">";
   650→        }
   651→    }
   652→    return "";
   653→}

   673→void HeadlessCLI::cmdPatch(const std::vector<std::string>& args) {
   674→    if (args.empty()) { err("usage: patch <list|apply|nop|save|undo|info>"); return; }
   675→    std::string sub = args[0];
   676→    if (sub == "list") {
   677→        printPatchList();
   678→    } else if (sub == "info") {
   679→        if (!patcher_.isLoaded()) { err("No target loaded. Use 'target <binary>'"); return; }
   680→        out("Target: " + patcher_.formatName());
   681→        out("Image size: " + std::to_string(patcher_.imageSize()) + " bytes");
   682→        out("Entry point: " + hex(patcher_.entryPoint()));
   683→        out("Sections: " + std::to_string(patcher_.sections().size()));
   684→    } else if (sub == "nop") {
   685→        if (args.size() < 3) { err("usage: patch nop <offset> <length>"); return; }
   686→        u64 off = strtoull(args[1].c_str(), nullptr, 0);
   687→        size_t len = strtoul(args[2].c_str(), nullptr, 0);
   688→        int id = patcher_.nopRange(off, len);
   689→        if (id < 0) err("patch failed: " + patcher_.lastError());
   690→        else out("Patch " + std::to_string(id) + ": NOPed " + std::to_string(len) + " bytes at " + hex(off));
   691→    } else if (sub == "apply") {
   692→        if (args.size() < 3) { err("usage: patch apply <offset> <kind> [bytes...]"); return; }
   693→        u64 off = strtoull(args[1].c_str(), nullptr, 0);
   694→        std::string kind = args[2];
   695→        BinaryPatcher::PatchKind k = BinaryPatcher::PatchKind::CustomBytes;
   696→        if (kind == "nop") k = BinaryPatcher::PatchKind::NOP;
   697→        else if (kind == "jmp") k = BinaryPatcher::PatchKind::JmpAlways;
   698→        else if (kind == "nojmp") k = BinaryPatcher::PatchKind::JmpNever;
   699→        else if (kind == "callnop") k = BinaryPatcher::PatchKind::CallToNop;
   700→        else if (kind == "rettrue") k = BinaryPatcher::PatchKind::RetTrue;
   701→        else if (kind == "ascii") k = BinaryPatcher::PatchKind::AsciiReplace;
   702→        size_t len = args.size() > 3 ? args.size() - 3 : 1;
   703→        std::vector<u8> bytes;
   704→        if (args.size() > 3) {
   705→            std::vector<std::string> btokens(args.begin() + 3, args.end());
   706→            bytes = parseBytes(btokens);
   707→            len = bytes.size();
   708→        }
   709→        int id = patcher_.applyPatch(off, k, len, bytes);
   710→        if (id < 0) err("patch failed: " + patcher_.lastError());
   711→        else out("Patch " + std::to_string(id) + " applied at " + hex(off));
   712→    } else if (sub == "save") {
   713→        if (args.size() < 2) { err("usage: patch save <outfile>"); return; }
   714→        if (patcher_.save(args[1])) out("Saved patched binary to " + args[1]);
   715→        else err("save failed: " + patcher_.lastError());
   716→    } else if (sub == "undo") {
   717→        // patch undo [id]  — if no id, undo the last applied patch
   718→        if (args.size() < 2) {
   719→            // Undo the last patch
   720→            auto& patches = patcher_.patches();
   721→            if (patches.empty()) { err("No patches to undo"); return; }
   722→            int last = (int)patches.size() - 1;
   723→            if (patcher_.undoPatch(last)) out("Undone patch " + std::to_string(last) + " (last)");
   724→            else err("undo failed");
   725→            return;
   726→        }
   727→        int id = atoi(args[1].c_str());
   728→        if (id < 0 || id >= (int)patcher_.patches().size()) {
   729→            err("Invalid patch id " + std::to_string(id) + " — valid range: 0.." +
   730→                std::to_string(patcher_.patches().size() - 1) + " (use 'patch list' to see IDs)");
   731→            return;
   732→        }
   733→        if (patcher_.undoPatch(id)) out("Undone patch " + std::to_string(id));
   734→        else err("undo failed");
   735→    } else {
   736→        err("Unknown patch subcommand: " + sub);
   737→    }
   738→}

   740→void HeadlessCLI::cmdStealth(const std::vector<std::string>& args) {
   741→    if (args.empty()) { printStealthStatus(); return; }
   742→    std::string sub = args[0];
   743→    if (sub == "list") {
   744→        printStealthStatus();
   745→    } else if (sub == "on" || sub == "off") {
   746→        if (args.size() < 2) { err("usage: stealth on|off <name>"); return; }
   747→        std::string nm = args[1];
   748→        for (auto t : AntiDetect::allTechniques()) {
   749→            std::string tn = AntiDetect::name(t);
   750→            // case-insensitive substring match
   751→            std::string lower_tn = tn, lower_nm = nm;
   752→            for (auto& c : lower_tn) c = tolower(c);
   753→            for (auto& c : lower_nm) c = tolower(c);
   754→            if (lower_tn.find(lower_nm) != std::string::npos) {
   755→                if (sub == "on") anti_.enable(t);
   756→                else anti_.disable(t);
   757→                out("Stealth: " + tn + " = " + (sub == "on" ? "ON" : "OFF"));
   758→                return;
   759→            }
   760→        }
   761→        err("No stealth technique matching '" + nm + "'");
   762→    } else {
   763→        err("usage: stealth [list|on <name>|off <name>]");
   764→    }
   765→}

   930→                  << hex(patches[i].file_offset, 8) << "       "
   931→                  << std::setw(4) << patches[i].length << "  ";
   932→        switch (patches[i].kind) {
   933→            case BinaryPatcher::PatchKind::NOP:         std::cout << "NOP"; break;
   934→            case BinaryPatcher::PatchKind::CustomBytes: std::cout << "BYTES"; break;
   935→            case BinaryPatcher::PatchKind::JmpAlways:   std::cout << "JMP"; break;
   936→            case BinaryPatcher::PatchKind::JmpNever:    std::cout << "NOJMP"; break;
   937→            case BinaryPatcher::PatchKind::CallToNop:   std::cout << "CALLNOP"; break;
   938→            case BinaryPatcher::PatchKind::RetTrue:     std::cout << "RETTRUE"; break;
   939→            case BinaryPatcher::PatchKind::AsciiReplace:std::cout << "ASCII"; break;

   941→        std::cout << "  " << patches[i].note << std::endl;
   942→    }

   979→
   980→int HeadlessCLI::run(int argc, char** argv, bool skip_eula) {
   981→    printBanner();
   982→    if (skip_eula) {
   983→        eula_accepted_ = true;
   984→    } else {
   985→        if (!showEula()) {
   986→            err("EULA declined. Exiting.");
   987→            return 1;
   988→        }
   989→        eula_accepted_ = true;
   990→    }
   991→
   992→    // Parse CLI args: --cli mode just enters REPL
   993→    bool batch_mode = false;
   994→    std::string batch_cmds;
   995→    for (int i = 1; i < argc; i++) {
   996→        std::string a = argv[i];
   997→        if (a == "-c" || a == "--commands") {
   998→            batch_mode = true;
   999→            if (i + 1 < argc) batch_cmds = argv[++i];
  1000→        } else if (a == "--no-eula-check") {
  1001→            // Already handled by skip_eula; ignore here
  1002→        } else if (a == "--help" || a == "-h") {
  1003→            std::cout << "Usage: ninjadb --cli [-c \"commands\"] [--no-eula-check]\n"
  1004→                      << "  --cli              Run in headless CLI mode\n"
  1005→                      << "  -c \"commands\"      Execute commands and exit (separated by ;)\n"
  1006→                      << "  --no-eula-check    Skip EULA acceptance prompt\n";
  1007→            return 0;
  1008→        }
  1009→    }
  1010→
  1011→    if (batch_mode) {
  1012→        std::istringstream ss(batch_cmds);
  1013→        std::string line;
  1014→        while (std::getline(ss, line, ';')) {
  1015→            if (!line.empty()) execute(line);
  1016→        }
  1017→        return 0;
  1018→    }
  1019→
  1020→    while (running_) {
  1021→        prompt();
  1022→        std::string line = readLine();
  1023→        if (line == "EOF") break;
  1024→        if (std::cin.eof()) break;
  1025→        if (line.empty()) continue;
  1026→        execute(line);
  1027→    }
  1028→    return 0;
  1029→}
